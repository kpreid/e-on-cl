(lambda (guard)
  (let ((sym (make-symbol (e-quote guard))))
    (setf (symbol-function sym) 
      (lambda (specimen)
        (eeq-is-same-yet (e. guard |coerce| specimen)
                         specimen)))
    `(satisfies ,sym)))
